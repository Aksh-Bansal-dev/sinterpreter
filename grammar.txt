program        → statement* EOF ;
statement      → exprStmt
               | printStmt ;

exprStmt       → comparison ";" ;
printStmt      → "print" comparison ";" ;
comparison           → term ( ( "<=" | ">=" | "<" | ">" | "==" || "!=" ) term )* ;
term           → factor ( ( "-" | "+" ) factor )* ;
factor         → unary ( ( "/" | "*" ) unary )* ;
unary          → ( "!" | "-" ) unary
               | primary ;
primary        → NUMBER | STRING | "true" | "false" | "nil"
               | "(" expression ")" ;
